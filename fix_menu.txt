   ĐÚNG:
   case 8: {
       int msnv = nhapSoNguyen("Nhập MSNV cần xóa: ");
       int pos = timMSNVTuanTu(ds, msnv);
       if (pos == -1) {
           printf(">> Không tìm thấy nhân viên có MSNV %d!\n", msnv);
       } else {
           printf("\n>> Thông tin nhân viên sẽ xóa:\n");
           xuat1NV(ds.a[pos]);
           printf("\nBạn có chắc muốn xóa? (1: Có | 0: Không): ");
           int xacNhan = nhapSoNguyen("");
           if (xacNhan == 1) {
               for (int i = pos; i < ds.n - 1; i++) {
                   ds.a[i] = ds.a[i + 1];
               }
               ds.n--;
               printf(">> Đã xóa nhân viên thành công!\n");
           } else {
               printf(">> Đã hủy thao tác xóa.\n");
           }
       }
       break;
   }

   }
